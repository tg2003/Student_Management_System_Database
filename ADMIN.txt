CREATE TABLE admin(
Admin_id CHAR(3),
Adm_name VARCHAR(100),
Adm_nic VARCHAR(11) UNIQUE,
Adm_adrs_no VARCHAR(6),
Adm_adrs_street VARCHAR(100),
Adm_adrs_city VARCHAR(20),
Adm_email VARCHAR(100) CHECK (Adm_email LIKE '%@%' AND Adm_email LIKE '%.%'),
Adm_dob DATE,
Adm_phone INT(10),
Dpt_name CHAR(3),
PRIMARY KEY(Admin_id)
);
