CREATE TABLE medical(
medical_id CHAR(6) NOT NULL,
status VARCHAR(20),
date_submitted DATE,
st_id CHAR(5) NOT NULL,
PRIMARY KEY(medical_id),
FOREIGN KEY(st_id) REFERENCES student(st_id)
);
