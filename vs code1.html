CREATE TABLE mark( st_id CHAR(5), exam_id CHAR(6), marks INT, FOREIGN KEY(exam_id) REFERENCES exam(exam_id), FOREIGN KEY(st_id) REFERENCES student(st_id) ); CREATE TABLE exam( exam_id CHAR(6) NOT NULL, exam_date DATE, exam_type VACHAR(10), total_marks INT, c_code CHAR(7), PRIMARY KEY(exam_id), FOREIGN KEY(c_code) REFERENCES course(c_code) ); CREATE TABLE session( session_id VARCHAR(20) NOT NULL, type CHAR(1), date DATE, c_code CHAR(5), PRIMARY KEY(session_id), FOREIGN KEY(c_code) REFERENCES course(c_code) ); CREATE TABLE attendence( date INT AUTO_INCREMENT, status VARCHAR(10), st_id CHAR(5), session_id CHAR(6), c_code CHAR(5), FOREIGN KEY(c_code) REFERENCES course(c_code), FOREIGN KEY(st_id) REFERENCES student(st_id), FOREIGN KEY(session_id) REFERENCES session(session_id) );
