CREATE TABLE marks( mark_id CHAR(4) NOT NULL, exam_id CHAR(6), st_id CHAR(5), marks INT, direct_marks INT, PRIMARY KEY(mark_id), FOREIGN KEY(exam_id) REFERENCES exam(exam_id), FOREIGN KEY(st_id) REFERENCES student(st_id) ); CREATE TABLE exam( exam_id CHAR(6) NOT NULL, c_code CHAR(7), exam_date DATE, exam_type VARCHAR, total_mark INT, PRIMARY KEY(exam_id), FOREIGN KEY(c_code) REFERENCES course(c_code) ); CREATE TABLE session( session_id VARCHAR(20) NOT NULL, c_code CHAR(5), week_no INT, type CHAR(1), date DATE, PRIMARY KEY(session_id), FOREIGN KEY(c_code) REFERENCES course(c_code) ); CREATE TABLE attendence( attendence_id INT AUTO_INCREMENT, c_code CHAR(5), st_id CHAR(5), session_id CHAR(6), status VARCHAR(10), PRIMARY KEY(attendence_id), FOREIGN KEY(c_code) REFERENCES course(c_code), FOREIGN KEY(st_id) REFERENCES student(st_id), FOREIGN KEY(session_id) REFERENCES session(session_id) );
