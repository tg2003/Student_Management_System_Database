    Dean_id CHAR(6) PRIMARY KEY,
    Name VARCHAR(50) NOT NULL,
    Email VARCHAR(50),
    Dob DATE,
    Nic CHAR(12),
    Phone VARCHAR(15),
    No VARCHAR(10),
    Street VARCHAR(50),
    City VARCHAR(50)
